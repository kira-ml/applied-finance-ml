# -----------------------------------------------------------------------------
# Core Data & Numerical Stack (Optimized for CPU/Memory)
# -----------------------------------------------------------------------------
# Pinning versions ensures bitwise reproducibility across different environments.
# These versions are selected for stability on Intel Core i5 architectures.

numpy==1.24.3
# Reason: Stable C-backend for vectorized threshold sweeping and matrix ops. 
# Version 1.24.x offers the best balance of performance and legacy compatibility for CPU-only laptops.

pandas==2.0.3
# Reason: Efficient DataFrame handling with built-in 'category' dtype support for RAM reduction.
# Includes optimized CSV parsing engines essential for quick ingestion on limited hardware.

scipy==1.10.1
# Reason: Required for statistical distributions in synthetic data generation (log-normal, etc.)
# and efficient sparse matrix operations if high-cardinality one-hot encoding is used.

# -----------------------------------------------------------------------------
# Machine Learning Engine
# -----------------------------------------------------------------------------

scikit-learn==1.3.0
# Reason: The sole ML dependency. Provides LogisticRegression (liblinear), 
# RandomForest, StandardScaler, and crucially, precision_recall_curve metrics.
# Version 1.3.0 includes optimized memory management for large sparse matrices.

# -----------------------------------------------------------------------------
# Configuration & Serialization
# -----------------------------------------------------------------------------

pyyaml==6.0.1
# Reason: Lightweight, no-compilation-required parser for config.yaml. 
# Enables the "Config-Driven" architecture without heavy framework overhead.

typing_extensions==4.9.0
# Reason: Backports typing.assert_never (Python 3.11+) for use on Python 3.10.

joblib==1.3.1
# Reason: Optimized for serializing large NumPy arrays and Scikit-Learn models.
# Faster and more compression-efficient than pickle for binary artifacts (.pkl).

# -----------------------------------------------------------------------------
# EXPLICIT EXCLUSIONS (Do Not Install)
# -----------------------------------------------------------------------------
# The following are strictly prohibited to maintain resource constraints:
# - tensorflow / torch : Overkill for tabular data; excessive RAM/CPU usage.
# - xgboost / lightgbm : Unnecessary complexity; native Sklearn RF/LogReg suffices.
# - flask / fastapi    : No microservices; inference is local function calls.
# - mlflow / wandb     : No cloud tracking; local JSON/Log files suffice.
# - airflow / prefect  : No orchestration; linear main.py script handles flow.
